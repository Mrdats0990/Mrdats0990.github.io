一.适配器Adapter【AccountAdapter.kt】
自定义的账户适配器类，用于在ListView中显示账户信息。适配器继承自BaseAdapter类，并重写了一些必要的方法。
在构造方法中，需要传入上下文Context和账户列表accountList。
重写的方法包括：
getCount(): 返回账户列表的数量。
getItem(): 根据指定位置返回对应的账户对象。
getItemId(): 根据指定位置返回对应的账户ID。
getView(): 根据指定位置创建或复用一个列表项的View，并为View设置账户的相关信息。同时，为列表项中的按钮设置点击事件监听器。
此外，适配器还定义了一些辅助方法：
deleteAccount(): 删除指定位置的账户，并更新数据源和视图。
raiseAccount(): 在指定位置的账户的余额上加1，并更新数据源和视图。
lowerAccount(): 在指定位置的账户的余额上减1，并更新数据源和视图。
值得注意的是，在代码中使用了AlertDialog来显示删除确认对话框，用户可以选择确认删除或取消操作。
在适配器中还引用了AccountDao类，用于操作账户数据的增删改查。可以根据具体需求实现其相关方法来完成数据的持久化操作。

class AccountAdapter (var content:Context,var accountList:MutableList<Account>): BaseAdapter() {
    lateinit var dao:AccountDao
    override fun getCount(): Int {
        return accountList.size
    }

    override fun getItem(p0: Int): Any {
        return accountList.get(p0)
    }

    override fun getItemId(p0: Int): Long {
        return p0.toLong()
    }

    override fun getView(p0: Int, p1: View?, p2: ViewGroup?): View {
        var view:View
        if(p1==null){
            view=LayoutInflater.from(content).inflate(R.layout.listview_item ,null)
        }else{
            view=p1
        }
        var account:Account=accountList.get(p0)
        var tv_id:TextView
        var tv_name:TextView
        var tv_balance:TextView
        var image_up:ImageView
        var image_down:ImageView
        var image_del:ImageView
        tv_id=view.findViewById(R.id.tv_id_listview)
        tv_name=view.findViewById(R.id.tv_name_listview)
        tv_balance=view.findViewById(R.id.tv_balance_listview)
        image_up= view.findViewById(R.id.image_up)
        image_down=view.findViewById(R.id.image_down)
        image_del=view.findViewById(R.id.image_delete)
        tv_id.setText(account.id.toString())
        tv_name.setText(account.name)
        tv_balance.setText(account.balance.toString())
        image_up.setOnClickListener {raiseAccount(p0)}
        image_down.setOnClickListener {lowerAccount(p0)}
        image_del.setOnClickListener { deleteAccount(p0) }
        return view
    }
    fun deleteAccount(id:Int){
        var dialog:AlertDialog.Builder=AlertDialog.Builder(content)
        var account=accountList[id]
        dialog.setTitle("警告")
        dialog.setIcon(R.mipmap.ic_launcher)
        dialog.setMessage("你真的要删除吗？")
        dialog.setPositiveButton("确定"){dialogInterface,i->

            var t:Int=dao.delete(account)
            if (t>0) {
                Toast.makeText(content, "数据删除成功", Toast.LENGTH_LONG).show()
            }else{
                Toast.makeText(content, "数据删除失败", Toast.LENGTH_LONG).show()
            }
                accountList.remove(account)
                notifyDataSetChanged()
            }
        dialog.setNegativeButton("取消"){dialogInterface,i->

        }
        dialog.show()
    }
    fun raiseAccount(id:Int){
        var account:Account=accountList.get(id)
        dao= AccountDao(content)
        account.balance=account.balance+1
        var t:Int=dao.update(account)
        notifyDataSetChanged()
    }
    fun lowerAccount(id:Int){
        var account:Account=accountList.get(id)
        dao= AccountDao(content)
        account.balance=account.balance-1
        var t:Int=dao.update(account)
        notifyDataSetChanged()
    }
}

二.数据访问层【AcountDao.kt】
用于对账户数据进行增删改查的操作。
在构造方法中，需要传入一个上下文Context对象，并初始化了数据库帮助类DataBaseHelper和账户列表accountList。
类中定义了一些方法，包括：
insert(): 向数据库中插入一个账户对象，通过ContentValues来设置账户的名称和余额，并返回插入后的账户ID。
getAllAccount(): 查询数据库中的所有账户数据，并将查询结果封装为一个Account对象列表。
update(): 更新数据库中指定账户的名称和余额。
delete(): 根据指定账户ID删除数据库中的对应账户。
lookAccountByName(): 根据指定名称模糊查询数据库中的账户数据，并返回符合条件的Account对象列表。
在每个操作方法中，都通过获取可读/可写的数据库实例，在使用完毕后，需要及时关闭Cursor和数据库连接。

相关代码：


class AccountDao (context: Context){
    lateinit var accountList: MutableList<Account>
    lateinit var db:SQLiteDatabase
    lateinit var helper:DataBaseHelper
    init {
        helper= DataBaseHelper(context)
        accountList= mutableListOf()
    }
    fun insert(account: Account){
        db=helper.writableDatabase
        val contentValue: ContentValues =ContentValues()
        contentValue.put("name",account.name)
        contentValue.put("balance",account.balance)
        var id:Long = db.insert("account",null,contentValue)
        account.id=id
    }
    fun getAllAccount():MutableList<Account>{
        db=helper.readableDatabase
        var account:Account
        var cursor:Cursor=db.query("account",null,null,null,null,null,null)
        while (cursor.moveToNext()){
            account= Account()
            var id:Long=cursor.getLong(cursor.getColumnIndex("_id"))
            var name:String=cursor.getString(cursor.getColumnIndex("name"))
            var balance:Int= cursor.getInt(cursor.getColumnIndex("balance"))
            account.id=id
            account.name=name
            account.balance=balance
            accountList.add(account)
        }
        cursor.close()
        return accountList
    }
    fun update(account: Account):Int{
        db=helper.writableDatabase
        var values:ContentValues= ContentValues()
        values.put("name",account.name)
        values.put("balance",account.balance)
        var t:Int=db.update("account",values,"_id=?", arrayOf(account.id.toString()))
        return t
    }
    //删除商品信息
    fun delete(account: Account):Int{
        db=helper.writableDatabase
        var t=db.delete("account","_id=?", arrayOf(account.id.toString()))
        return t
    }
    fun lookAccountByName(name:String):MutableList<Account> {
        db = helper.readableDatabase
        var cursor: Cursor =
            db.query("account", null, "name like ?", arrayOf("%" + name + "%"), null, null, null)
            accountList.clear()
        var account:Account
        var name:String
        var id:Long
        var balance:Int
        while (cursor.moveToNext()){
            account= Account()
            id=cursor.getLong(cursor.getColumnIndex("_id"))
            name=cursor.getString(cursor.getColumnIndex("name"))
            balance=cursor.getInt(cursor.getColumnIndex("balance"))
            account.let {
                it.id=id
                it.name=name
                it.balance=balance
            }
            accountList.add(account)
        }
        return accountList
    }
}
